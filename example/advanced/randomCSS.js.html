    var values={
        bgr:getRand(),
        bgg:getRand(),
        bgb:getRand(),
        tr:getRand(),
        tg:getRand(),
        tb:getRand(),
    }

    var vars=Object.keys(values);

    for(var i in vars){
        var reg=new RegExp(
            '\{\{'+vars[i]+'\}\}',
            'g'
        );

        body.value=body.value.replace(
            reg,
            values[
                vars[i]
            ]
        );
    }
}

function getRand(){
    return Math.round(
        Math.random()*255
    );
}
